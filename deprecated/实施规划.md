```
输出：(angle, turns, predict_x, predict_y) 或 None

1. tx, ty = tgt.x, tgt.y  # 初始猜测：目标当前位置
2. for _ in range(8):      # 迭代收敛
3.     angle, turns = get_arrival(src.x, src.y, tx, ty, ships)
4.     ntx, nty = predict_pos(tgt, ..., turns)
5.     if pos is None: return None     # 彗星过期
6.     if dist(ntx, nty, CENTER) < 22: return None  # 近日点禁区
7.     if |ntx - tx| < 0.2 and |nty - ty| < 0.2: break  # 收敛
8.     tx, ty = ntx, nty
9. angle, turns = get_arrival(src.x, src.y, tx, ty, ships)
10. return (angle, turns, tx, ty)
```

### 4.2 特征设计

#### `src/features/self_features.py` — 源行星特征（21 → 最终 21 维）

核心改进：显式编码飞行中舰队信息，让 agent 知道每颗己方行星是否正被攻击。

```python
def build_self_features(src: PlanetState, state: GameState,
                        ledger: dict, timeline: dict) -> np.ndarray:
    t = timeline.get(src.id, {})
    incoming = ledger.get(src.id, [])  # [(eta, owner, ships), ...]

    # ── 飞行中舰队：谁正飞来这颗行星？（5 维）── ✨ 新增
    my_incoming = sum(s for eta, own, s in incoming if own == state.player)
    enemy_incoming = sum(s for eta, own, s in incoming if own != state.player and own != -1)
    neutral_incoming = sum(s for eta, own, s in incoming if own == -1)  # 彗星/中立

    enemy_etas = [eta for eta, own, s in incoming if own != state.player and own != -1]
    first_enemy_eta = min(enemy_etas) if enemy_etas else 999

    # ── 可达性（用 ETA 替代原始距离）──
    # 找出从其他行星出发到达本行星的最短 ETA
    min_enemy_eta_to_me = min(
        (travel_time(p.x, p.y, p.radius, src.x, src.y, src.radius, max(1, p.ships))
         for p in state.enemy_planets if p.id != src.id),
        default=999
    )
    min_neutral_eta_to_me = min(
        (travel_time(p.x, p.y, p.radius, src.x, src.y, src.radius, max(1, p.ships))
         for p in state.neutral_planets if p.id != src.id),
        default=999
    )

    return [
        # ── 基本属性（6 维）──
        src.x / BOARD_SIZE,
        src.y / BOARD_SIZE,
        src.radius / 5.0,
        min(src.ships, 400) / 400,
        src.production / 5.0,
        float(is_rotating(src, state)),

        # ── 空间特征（3 维）── ✨ 用 ETA 替代原始距离
        dist_to_sun(src) / 70.0,
            # 太阳距离保留：影响遮挡概率和轨道特性（非 ETA 可替代）
        1.0 / (1.0 + min_enemy_eta_to_me),
            # 最近敌方多快能到我这里（ETA 编码，非距离）
        1.0 / (1.0 + min_neutral_eta_to_me),
            # 最近中立行星多快能到我这里

        # ── 防御态势（5 维）──
        defense_risk,
        float(has_sun_block_to_nearest_enemy),
        float(has_sun_block_to_nearest_neutral),
        enemy_incoming / max(1.0, src.ships + src.production * max(1, first_enemy_eta)),
        1.0 / (1.0 + first_enemy_eta),

        # ── 增援状态（3 维）──
```

#### `src/features/global_features.py` — 全局特征（16 维）

核心原则：不编码"天上有多少船"（无意义——船可能打不中、可能飞向无关目标），
而是编码"哪些行星正被瞄准 + 最危急局面 + 特殊事件"。

```python
def build_global_features(state: GameState, ledger: dict) -> np.ndarray:
    # ── 有多少颗我方行星正被敌方舰队瞄准？──
    my_planets_under_attack = sum(
        1 for pid in ledger
        if any(own != state.player and own != -1 for _, own, _ in ledger[pid])
        and any(p.id == pid and p.owner == state.player for p in state.my_planets)
    )

    # ── 最危急防御点的压力比 ──
    max_defense_pressure = 0.0
    for p in state.my_planets:
        incoming = ledger.get(p.id, [])
        enemy_ships = sum(s for _, own, s in incoming if own != state.player and own != -1)
        if enemy_ships <= 0:
            continue
        enemy_etas = [eta for eta, own, _ in incoming if own != state.player and own != -1]
        first_eta = min(enemy_etas)
        defense_capacity = p.ships + p.production * first_eta
        pressure = enemy_ships / max(1.0, defense_capacity)
        max_defense_pressure = max(max_defense_pressure, pressure)
    # ≥ 1.0 意味着至少一颗行星确定失守

    # ── 有多少颗敌方行星已被我方舰队瞄准？──
    enemy_planets_targeted = sum(
        1 for pid in ledger
        if any(own == state.player for _, own, _ in ledger[pid])
        and any(p.id == pid and p.owner not in (-1, state.player) for p in state.planets)
    )

    # ── 行星上驻军（不含飞行中舰队）──
    my_garrison = sum(p.ships for p in state.my_planets)
    enemy_garrison = sum(p.ships for p in state.enemy_planets)
    my_max_prod = max((p.production for p in state.my_planets), default=0)
    enemy_max_prod = max((p.production for p in state.enemy_planets), default=0)

    return [
        # ── 回合进度（2 维）──
        state.step / 500.0,
        1.0 - state.step / 500.0,

        # ── 版图（3 维）──
        len(state.my_planets) / 48,
        len(state.enemy_planets) / 48,
        len(state.neutral_planets) / 48,

        # ── 经济（2 维）──
        state.my_total_production / (48 * 5),
        state.enemy_total_production / (48 * 5),

        # ── 军力（3 维）──
        state.my_total_ships / 19200,
        state.enemy_total_ships / 19200,
        (state.my_total_ships - state.enemy_total_ships)
            / max(1, state.my_total_ships + state.enemy_total_ships),

        # ── 攻防态势 & 特殊事件（6 维）── ✨ 全面重构
        my_planets_under_attack / max(1, len(state.my_planets)),
            # 防线被突破的广度
        max_defense_pressure,
            # 最危急行星的压力比（≥1.0 = 确定失守，网络学会"尖叫求救"）
        enemy_planets_targeted / max(1, len(state.enemy_planets)),
            # 进攻广度
        float(len(state.comet_ids) > 0),
            # 是否有彗星（game-changer，需要切换全局策略）
        my_max_prod / max(1.0, enemy_max_prod),
            # 最强行星的产量对比（经济集中度）
        my_garrison / max(1.0, enemy_garrison),
            # 行星驻军对比（排除飞行舰队，衡量"防守姿态"）
    ]
```

## 5. 第三阶段：动作空间设计

### 5.1 目标

同时学习 **"打哪个目标"** 和 **"派多少舰船"**。

### 5.2 动作空间

```
action = (target_index: int, ship_ratio: float ∈ [0, 1])
```

- `target_index`：在决策矩阵中选择第 i 行（含 no-op 行，candidate_id=-1）
- `ship_ratio`：发送「可用舰船 × ratio」艘（clip 到 [MIN_FLEET, available]）
- 每颗己方行星独立决策，同一回合内可派多支舰队

**no-op 表示：** builder 为每个源行星额外生成一条 `candidate_id=-1` 的特殊行，
cand_feat=全零。策略通过正常的离散选择机制比较 no-op 与真实目标，无需额外机制。

**多行星并发：** 初始版本独立处理每颗行星（不同行星间不共享 reserve 调整）。
若后续出现过度派兵问题，再引入全局 reserve 协调。

### 5.3 实现方案

采用 **per-candidate 共享权重** 架构（与 §7.1 网络结构一致），
而非固定 K 的大矩阵方案（后者无法处理变长候选列表）。

使用 **Beta 分布** 对 `ship_ratio` 建模（天然有界 [0,1]，比 Gaussian + tanh 更稳定）。
ship_alpha/beta 从 value_input（所有候选的池化表示）计算，即 **ship_ratio 是源行星级别的全局决策**，
不依赖具体选了哪个目标——这降低了复杂度，且 agent 可通过"选弱目标时降低 ratio 的联合概率"
隐式学习目标-ratio 的适配。

```python
# src/policy/action_head.py

class TargetHead(nn.Module):
    """Per-candidate 离散 logit: hidden_dim → 1"""
    def __init__(self, hidden_dim):
        super().__init__()
        self.fc = nn.Linear(hidden_dim, 1)

    def forward(self, joint):  # joint: (K, hidden_dim)
        return self.fc(joint).squeeze(-1)  # → (K,)


class ShipAlphaHead(nn.Module):
    """Beta 分布 α 参数: hidden_dim → 1"""
    def __init__(self, hidden_dim):
        super().__init__()
        self.fc = nn.Linear(hidden_dim, 1)

    def forward(self, value_input):  # value_input: (hidden_dim,)
        return F.softplus(self.fc(value_input)) + 1.0  # 确保 > 1


class ShipBetaHead(nn.Module):
    """Beta 分布 β 参数: hidden_dim → 1"""
    def __init__(self, hidden_dim):
        super().__init__()
        self.fc = nn.Linear(hidden_dim, 1)

    def forward(self, value_input):
        return F.softplus(self.fc(value_input)) + 1.0


class ValueHead(nn.Module):
    """状态价值估计: hidden_dim → 1"""
    def __init__(self, hidden_dim):
        super().__init__()
        self.fc = nn.Linear(hidden_dim, 1)

    def forward(self, value_input):
        return self.fc(value_input).squeeze(-1)


def sample_action(target_logits, ship_alpha, ship_beta, mask=None,
                  deterministic=False):
    """从策略分布中采样动作。
    
    Args:
        target_logits: (K,) 每个候选的 logit
        ship_alpha: scalar Beta α
        ship_beta: scalar Beta β
        mask: (K,) bool, True=有效, False=屏蔽
        deterministic: True=取 argmax/mean
    
    Returns:
        (target_idx, ship_ratio, target_log_prob, ship_log_prob)
    """
    if mask is not None:
        target_logits = target_logits.clone()
        target_logits[~mask] = float('-inf')
    
    target_dist = Categorical(logits=target_logits)
    target_idx = target_dist.probs.argmax() if deterministic else target_dist.sample()
    
    beta_dist = Beta(ship_alpha, ship_beta)
    ship_ratio = beta_dist.mean if deterministic else beta_dist.sample()
    ship_ratio = ship_ratio.clamp(1e-6, 1.0 - 1e-6)
    
    return (target_idx, ship_ratio,
            target_dist.log_prob(target_idx), beta_dist.log_prob(ship_ratio))


def compute_ships_to_send(available, ship_ratio, min_fleet=12):
    """将 ship_ratio 转换为实际舰船数。"""
    raw = int(available * ship_ratio)
    if raw < min_fleet:
        return 0
    return min(raw, available)
```

## 6. 第四阶段：奖励塑形

### 6.1 设计哲学：统一的势能函数 Φ（Future-Value-First）

游戏规则的胜利条件是：**终局时，行星驻军 + 舰队舰船总数最大者获胜。**

由此推导出一条核心原则：

> **奖励函数只回答一个问题——"你现在离胜利更近了还是更远了？"**
> 所有中间奖励必须是终局胜利条件的**单调代理信号**，不能是启发式猜测。

以及一个关键的对称性洞察：

> **敌人的收益就是我的损失，反之亦然。不存在独立的"惩罚逻辑"——**
> 同一个势能函数 Φ(s, player_id) 定义每个玩家的未来价值，
> ΔΦ = Φ(s') − Φ(s) 自动同时处理奖励和惩罚。

**为什么需要对称 Φ？** 旧方案的致命不对称在于只看我方所有权变化：
敌人占领一颗产能 5、寿命 400 的中立行星 → 我方奖励信号 = 0（盲区）。但从"离胜利更近还是更远"的角度，敌人扩张明确意味着我方相对胜率下降。对称 Φ 一劳永逸地解决了这个问题。

```python
def state_potential(state, player_id) -> float:
    """零和势能：我方领土现值 − 敌方领土现值"""
    my_pv = sum(normalize_pv(capture_present_value(p, state))
                for p in state.planets if p.owner == player_id)
    enemy_pv = sum(normalize_pv(capture_present_value(p, state))
                   for p in state.planets if p.owner not in (-1, player_id))
    return my_pv - enemy_pv
```

**奖励 = Φ(s') − Φ(s)**——纯 PBRS，一行代码覆盖所有权变化的全部六种情况：

| 事件 | Δ(my_PV) | Δ(enemy_PV) | ΔΦ | 说明 |
|------|-----------|-------------|-----|------|
| 我占中立 | +pv | 0 | **+pv** | 标准收益 |
| 敌占中立 | 0 | +pv | **−pv** | 自动惩罚！敌人扩张→相对劣势 |
| 我抢敌星 | +pv | −pv | **+2pv** | 抢夺天然两倍权重 |
| 敌抢我星 | −pv | +pv | **−2pv** | 失利天然两倍惩罚 |
| 我失中立（彗星过期） | −pv | 0 | **−pv** | 损失 |
| 敌失中立（彗星过期） | 0 | −pv | **+pv** | 自动奖励！敌人损失→相对优势 |

**关键优势：**
- **零惩罚逻辑**：不存在 if-else 分支来区分"奖励"和"惩罚"——ΔΦ 自然为正或负。
- **可复用**：同一个 `state_potential()` 函数可以用来评估**任意玩家在任意状态的价值**，为未来 4 人乱斗 / 对手建模打下基础。
- **可扩展**：只需替换 Φ 的定义（如 `Φ = my_PV − max(others_PV)`），即可从 1v1 切换到多人模式，上层 `ΔΦ` 奖励逻辑完全不变。

### 6.2 剩余寿命计算

```python
def planet_remaining_life(planet, state: GameState) -> int:
    """行星还能产出多少回合。彗星用实际剩余寿命，普通行星用游戏剩余回合。"""
    if planet.id in state.comet_ids:
        return comet_remaining_life(planet.id, state.comets)
    else:

```

### 6.3 占领现值（Present Value）

```python
def capture_present_value(planet, state: GameState) -> float:
    """占领行星的未来总产出（原始值，最大 = 500×5 = 2500）。"""
    return planet_remaining_life(planet, state) * planet.production

def normalize_pv(raw_pv: float) -> float:
    """归一化到 [0, 5] 范围，与 Δ舰船优势量级匹配。"""
    return raw_pv / 500.0   # 2500/500 = 5.0 (max)
```

#### 关键游戏机制影响

1. **彗星过期在舰队发射之前处理**：占领者在彗星离开的最后一回合**无法从中派出舰队**。驻军在过期时直接消失。

2. **已占领彗星过期 = 资产蒸发**：`planet_remaining_life()` 正确返回 0 → `capture_present_value()` 返回 0 → Φ 自动处理（`my_PV` 下降 → ΔΦ < 0）。

3. **4 人模式下彗星更稀缺**：仅 2 波（vs 1v1 的 5 波），但每波和 1v1 完全相同。彗星作为跳板的价值在 4 人模式下更高——少了 3 波彗星，错过了就是错过了。

#### 对奖励函数的启示

- **彗星 PV 验证**：产能固定 1，存留 5~40 回合 → 归一化 PV = 0.01 ~ 0.08（本测试种子 32 回合 → 0.064）。比普通高产星（PV 可达 5.0）低两个数量级。
- **warmup 必要性再次确认**：Φ 对彗星几乎给不出信号（PV < 0.08），但彗星→敌军跳板→抢夺的奖励链价值可达 2×普通行星 PV（~3~10）。warmup 的 1.5 额外加分恰好弥补了这个 Huge 信号差距。
- **4 人模式零修改可用**：现有的 `comet_remaining_life()`、`capture_present_value()`、`state_potential()` 在 4 人模式下完全不需要改动——引擎的彗星数据结构在两个模式间完全一致。

若将来发现 agent 过于保守（不敢抢敌星），可以小幅上调 α 到 1.05 ~ 1.1。但初始值 1.0 已经是理论最优——Φ 的对称性给出了正确的相对信号。

### 6.6 彗星战略 warmup（自适应探索引导）

**问题：** 彗星自身 PV 极低（产能固定为 1、棋盘存留仅 5~40 回合 → 归一化 PV ≈ 0.01 ~ 0.08），对称 Φ 也给不出强信号。
但彗星的真正价值在于作为前线跳板，后续从彗星出发突袭敌方时 +2pv 大奖才会兑现。
这条信用分配链长达 25+ 回合，超过 GAE 有效窗口（~17 回合）。
随机探索几乎不可能触发"占彗星 AND 从彗星突袭成功"的联合事件——agent 大概率永远学不会碰彗星。

**核心原则：学会才退，没学会一直引导，退场后零残留。**

#### 机制设计

warmup 不由固定局数控制，而是跟踪一个指标：**彗星利用率**——agent 是否会把已占领的彗星当作攻击跳板？

```python
# src/env/reward.py

COMET_WARMUP_MAX = 1.5           # 彗星占领最大额外加分
TARGET_UTILIZATION = 0.20        # 目标：20% 的 episode 中有彗星→敌星攻击


def get_comet_warmup(utilization_ema: float) -> float:
    """根据彗星利用率 EMA 计算 warmup 值。

    utilization_ema = 0.0  →  warmup = 1.50（全开，强引导）
    utilization_ema = 0.10 →  warmup = 0.75（半开）
    utilization_ema ≥ 0.20 →  warmup = 0.00（退场，值函数已自举）
    """
    ratio = min(1.0, utilization_ema / TARGET_UTILIZATION)
    return COMET_WARMUP_MAX * (1.0 - ratio)
```

**训练循环中每局更新 utilization EMA：**

```python
# src/train.py 训练循环内

comet_utilization_ema = 0.0  # 初始值

for episode in range(total_episodes):
    # ... 运行一局 ...
    attacked_from_comet = did_agent_attack_from_comet_this_episode()  # 0 or 1

    comet_utilization_ema = 0.99 * comet_utilization_ema + 0.01 * attacked_from_comet
    warmup = get_comet_warmup(comet_utilization_ema)

    # warmup 作为参数注入 reward 函数
    reward = compute_step_reward(..., comet_warmup=warmup)
```

**判定函数：** 本局内 agent 是否选择过"源行星是已占领彗星、目标是敌方行星"的动作？是 → 1，否 → 0。

#### 为什么"绝对学会"

| 场景 | utilization EMA | warmup | 效果 |
|------|----------------|--------|------|
| 训练初期，agent 从未从彗星攻击 | 0.00 | **1.50** | 全开引导 |
| 偶尔发现链条，开始尝试 | 0.05 | **1.125** | 开始衰减 |
| 行为稳定，经常利用彗星 | 0.15 | **0.375** | 接近退场 |
| 熟练掌握，彗星=常规跳板 | ≥0.20 | **0.00** | 完全退场 |
| （安全网）退场后行为退化 | 回落到 <0.20 | **自动恢复** | warmup 重新激活 |

warmup 不是定时炸弹——不存在"5000 局到了还没学会"的问题。学不会就一直引导，学会了自动归零，退化了自己复活。

#### warmup 与对称 Φ 的兼容性

- warmup 只在训练阶段通过 `comet_warmup` 参数注入 `compute_step_reward()`
- 它仅作用于 **中立→我方的彗星占领**：`ΔΦ + warmup`（而 ΔΦ 仅给 ~0.06，warmup 给 1.5 → 合计 ~1.56）
- warmup 收敛到 0 后，彗星占领回归纯 PBRS 信号，交给值函数自举
- warmup 和 α 互不竞争——α 影响的是 Φ 内部的领土权重，warmup 是 Φ 之外的探索引导

### 6.7 完整奖励函数

```python
# src/env/reward.py

def compute_step_reward(prev_state: GameState, curr_state: GameState,
                        player: int, game_ended: bool = False,
                        comet_warmup: float = 0.0) -> float:
    """
    三层奖励结构（对称 Φ 框架）：
      Layer 1: Δ舰船优势 —— 比值变化 × 3.0（对称，天然编码代价信号）
      Layer 2: ΔΦ_territory —— 领土势能变化（一行覆盖全部六种所有权变化情况）
      Layer 3: 终局 ±10.0
      + comet_warmup: 彗星探索引导加分（训练阶段外部注入，评估时为 0）
    """
    reward = 0.0

    # ═══════════════════════════════════════════════════════
    # Layer 1: 舰船优势变化（代价信号）
    # ═══════════════════════════════════════════════════════
    def ship_ratio(state, owner):
        my = (sum(p.ships for p in state.planets if p.owner == owner) +
              sum(f.ships for f in state.fleets if f.owner == owner))
        enemy = (sum(p.ships for p in state.planets if p.owner == (1 - owner)) +
                 sum(f.ships for f in state.fleets if f.owner == (1 - owner)))
        return my / max(1.0, my + enemy)

    prev_ratio = ship_ratio(prev_state, player)
    curr_ratio = ship_ratio(curr_state, player)
    reward += 3.0 * (curr_ratio - prev_ratio)

    # ═══════════════════════════════════════════════════════
    # Layer 2: 领土势能变化 ΔΦ（价值信号 + 惩罚信号，一体两面）
    # ═══════════════════════════════════════════════════════
    prev_phi = state_potential(prev_state, player)
    curr_phi = state_potential(curr_state, player)
    reward += (curr_phi - prev_phi)

    # ═══════════════════════════════════════════════════════
    # comet_warmup: 彗星探索引导（独立于 PBRS，训练阶段注入）
    # ═══════════════════════════════════════════════════════
    for p in curr_state.planets:
        prev = _find_planet(prev_state, p.id)
        if prev is None:
            continue
        # 刚占领的彗星 → 额外加分（仅训练阶段 comet_warmup > 0 时生效）
        if (prev.owner != player and p.owner == player
                and p.id in curr_state.comet_ids):
            reward += comet_warmup

    # ═══════════════════════════════════════════════════════
    # Layer 3: 终局奖励
    # ═══════════════════════════════════════════════════════
    if game_ended:
        my_score = (sum(p.ships for p in curr_state.planets if p.owner == player) +
                    sum(f.ships for f in curr_state.fleets if f.owner == player))
        enemy_score = (sum(p.ships for p in curr_state.planets if p.owner == (1 - player)) +
                       sum(f.ships for f in curr_state.fleets if f.owner == (1 - player)))
        if my_score > enemy_score:
            reward += 10.0
        elif my_score < enemy_score:
            reward -= 10.0
        # 平局: 0

    return reward
```

**为什么要保留 comet_warmup 的逐行星扫描？** 虽然 Φ 已经自动编码了所有权变化，但彗星的 PV 天然极低（产能固定 1，存留 5~40 回合 → PV ≈ 0.01~0.08），ΔΦ 给的信号太弱。warmup 需要一个明确的"侦测到彗星被占领"的触发点来叠加。
这个扫描不影响 ΔΦ 的其他部分——它是叠加在 Φ 之上的纯加法项。

### 6.8 设计验证：场景走查

| 场景 | ΔΦ | Δship | comet_warmup | 净奖励 | 结论 |
|------|-----|--------|-------------|--------|------|
| 占领产能4中立星（T=50，损15艘，总舰~2000） | +3.6 | −0.02 | — | **+3.58** | 强烈正信号 |
| 亏损占领产能1敌星（T=400，损80艘） | ≈0 | −0.24 | — | **≈−0.24** | 亏损被惩罚 |
| 抢夺敌刚占的产能4星（T=100，仅损8艘） | +7.2 | −0.01 | — | **+7.19** | 最强烈正信号（2×PV 天然对称） |
| 敌占产能4中立星（T=100） | **−3.6** | 0 | — | **−3.60** | ✨ 新增：自动惩罚！原方案为 0（盲区） |
| 敌彗星过期（敌损失资产） | **+0.06** | 0 | — | **+0.06** | ✨ 新增：自动奖励！原方案为 0（盲区） |
| 彗星（产能1，余命30，损10艘，训练初期） | +0.06 | −0.02 | +1.50 | **+1.54** | warmup 强引导探索 |
| 彗星（产能1，余命30，损10艘，训练后期） | +0.06 | −0.02 | 0 | **+0.04** | warmup 已退场，依赖值函数 |

**关键设计验证：**
- 彗星寿命短 → PV 天然低 → warmup 训练初期补足探索信号，逐渐退场交棒给值函数
- 后期占领 → 剩余回合少 → PV 自动衰减
- 低产能行星 → PV 低（不值得花大代价）
- 抢夺场景 → 对称 Φ 天然给 2×PV（不再需要 α > 1 来特殊处理）
- 敌人扩张 → 首次被正确纳入奖励信号（原方案盲区）
- warmup 仅影响彗星占领，与 Φ 的其余路径互不干扰

#### 需要适配的部分

唯一需要改动的是 Φ 的定义（`state_potential`），从零和改为多人势能：

```python
def state_potential_multiparty(state, player_id) -> float:
    """多人势能：我方 PV − 最强对手的 PV（盯住第一名）"""
    my_pv = sum(normalize_pv(capture_present_value(p, state))
                for p in state.planets if p.owner == player_id)
    others = set(owner for p in state.planets if p.owner not in (-1, player_id))
    if not others:
        return my_pv
    best_enemy_pv = max(
        sum(normalize_pv(capture_present_value(p, state))
            for p in state.planets if p.owner == other)
        for other in others
    )
    return my_pv - best_enemy_pv
```

同一套 `compute_step_reward()` 的 `ΔΦ` 逻辑完全不变，只换 Φ 定义即可适配。

## 7. 第五阶段：策略网络设计

### 7.1 架构

```
                   ┌─────────────────────────────┐
self_feat (21)  ──→│ SelfEncoder                 │──→ 256d
                   │ Linear(21→256) → ReLU       │
                   │ Linear(256→256) → ReLU      │
                   └─────────────────────────────┘
                                                    ┌──→ TargetHead → 1 (per-candidate)
                   ┌─────────────────────────────┐  │
cand_feat (K×29) ─→│ CandidateEncoder (共享权重)  │──→ 256d ──┤
                   │ Linear(29→256) → ReLU       │  │         ├──→ ShipAlphaHead → 1
                   │ Linear(256→256) → ReLU      │  │         │
                   └─────────────────────────────┘  │         ├──→ ShipBetaHead → 1
                                                    │         │
                   ┌─────────────────────────────┐  │         └──→ ValueHead → 1
global_feat (16) ─→│ GlobalEncoder               │──→ 256d
                   │ Linear(16→256) → ReLU       │
                   │ Linear(256→256) → ReLU      │
                   └─────────────────────────────┘

拼接方式：
  对每个候选 i：
    joint[i] = concat(self_hidden, global_hidden, cand_hidden[i])  # 768d
    target_logit[i] = TargetHead(joint[i])                         # 768→1, 共享权重

  pooled_cand = mean(cand_hidden, dim=1)  # 256d
  value_input = concat(self_hidden, global_hidden, pooled_cand)  # 768d
  value = ValueHead(value_input)            # 768→1
  ship_alpha = ShipAlphaHead(value_input)   # 768→1
  ship_beta = ShipBetaHead(value_input)     # 768→1
```

### 7.2 关键设计决策

| 决策 | 选择 | 理由 |
|------|------|------|
| CandidateEncoder 共享权重 | ✅ | 候选之间对称，共享参数降低过拟合 |
| 隐藏层大小 | 256（比 PPO 教程的 128 大一倍） | 特征更多，需要更大容量 |
| 激活函数 | ReLU | 简单稳定 |
| 初始化 | Orthogonal（PPO 标准） | 改善训练初期稳定性 |
| ship_alpha/beta 从 value_input 出 | ✅ | ship_ratio 是全局决策，不应依赖单个候选 |

### 8.3 混合动作 PPO 更新

```python
def ppo_update(policy, optimizer, batch, ...):
    # batch 包含：
    #   target_index, target_log_prob
    #   ship_ratio, ship_log_prob
    #   returns, advantages

    for epoch in range(epochs):
        outputs = policy(self_feat, cand_feat, global_feat, mask)

        # 离散部分
        new_target_log_prob = Categorical(logits=outputs.target_logits).log_prob(target_index)

        # 连续部分
        new_ship_log_prob = Beta(outputs.ship_alpha, outputs.ship_beta).log_prob(ship_ratio)

        # 总 log_prob
        new_log_prob = new_target_log_prob + new_ship_log_prob
        old_log_prob = batch.target_log_prob + batch.ship_log_prob

        ratio = (new_log_prob - old_log_prob).exp()

        # PPO clip loss
        policy_loss = -torch.min(
            advantages * ratio,
            advantages * torch.clamp(ratio, 1 - clip_coef, 1 + clip_coef)
        ).mean()

        value_loss = 0.5 * (returns - outputs.value).pow(2).mean()

        # 离散 + 连续熵
        entropy = target_entropy + ship_entropy

        loss = policy_loss + vf_coef * value_loss - ent_coef * entropy

        optimizer.zero_grad()
        loss.backward()
        torch.nn.utils.clip_grad_norm_(policy.parameters(), max_grad_norm)
        optimizer.step()
```
